"pinionToothCount" = 20
"gearToothCount" = 40
"pressureAngle" = 20deg
"module" = 1mm
"faceWidth" = 7.5mm
"toothCount" = "pinionToothCount"
"pitchDiam" = "module" * "toothCount"
"pitchConeAngle" = IIF("toothCount"  = "pinionToothCount", atn("pinionToothCount" / "gearToothCount"), atn("gearToothCount" / "pinionToothCount"))
"addendumAngle" = atn(2 * sin("pitchConeAngle") / "toothCount")
"dedendumAngle" = atn(2 * 1.25 * sin("pitchConeAngle") / "toothCount")
"virtPitchDiam" = "pitchDiam" / cos("pitchConeAngle")
"virtBaseDiam" = "pitchDiam" * cos("pressureAngle") / cos("pitchConeAngle")
"virtRootDiam" = ("pitchDiam" - 2.5 * "module" * cos("pitchConeAngle")) / cos("pitchConeAngle")
"virtAddendumDiam" = ("pitchDiam" + 2 * "module" * cos("pitchConeAngle")) / cos("pitchConeAngle")
"halfToothAngle" = ((PI - "pressureAngle" * PI / 180) / (2 * "toothCount")) * cos("pitchConeAngle")
"tmax" = 1.25 * sqr("virtAddendumDiam"^2 / "virtBaseDiam"^2 - 1)
