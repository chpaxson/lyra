"m" = 1
"z" = 20 ' tooth count
"delta" = 60 ' pitch angle
"dp" = "m" * "z" ' pitch diameter

"L" = 7.5 ' tooth length
"a" = 20 ' angle of action
"tha" = atn(2 * sin("delta") / "z") ' addendum angle
"thf" = atn(2 * 1.25 * sin("delta") / "z") ' dedendum angle
"dpv" = "dp" / cos("delta") ' virtual pitch diameter
"dbv" = "dp" * cos("a") / cos("delta") ' virtual base diameter
"dfv" = ("dp" - 2.5 * "m" * cos("delta")) / cos("delta") ' virtual root diameter
"dav" = ("dp" + 2 * "m" * cos("delta")) / cos("delta") ' virtual addendum diameter
"dfvu" = IIF("dbv" > "dfv", "dfv", "dbv" * 0.99) ' dfv undercut
"psi" = 360 / (4 * "z" / cos("delta")) ' half tooth angle
"inva" = tan("a") * 180 / pi - "a" ' involute of alpha
"tmax" = sqr("dav"^2 / "dbv"^2 - 1) * 1.1 ' max involute param
